#!/bin/bash

if [ $# -ne 1 ]; then
    echo "Использование: $0 <текстовый параметр>"
    exit 1
fi

PARAM="$1"

# Проверяем, является ли параметр числом
if [[ "$PARAM" =~ ^[0-9]+$ ]]; then
    echo "Ошибка: параметр не должен быть числом"
    exit 1
fi

# Выводим значение параметра
echo "Введённый параметр: $PARAM"
